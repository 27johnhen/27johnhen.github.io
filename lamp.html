lamp.ino Arduino code and wiring

        
 int d = 100;

 void setup(){
   pinMode(9, OUTPUT);
   Serial.begin(9600);
   while (!Serial);
   Serial.println("Send 'n' to turn lamp on. Send 'f' to turn lamp off.");
 }

 void loop() { 
   if (Serial.available()){
     char state = (char) Serial.read();
     if (state == 'n'){
       digitalWrite(9, HIGH);
       Serial.println("lamp on");
     }
     if (state == 'f'){
       digitalWrite(9, LOW);
       Serial.println("lamp off");
     }
     if (state == 's'){
       d=20;
     }
     if (state == 'l'){
       d=200;
     }
     if (state == 'x'){
       d=100;
     }
   }
   delay(d);
 }
Wiring: Connect pin 9 of an Arduino board to the signal pin of a relay; connect the relay's power and ground pins to the power and ground pins of the Arduino board. Connect a light and power supply to the relay. The light circuit attached to the relay should be normally open.
